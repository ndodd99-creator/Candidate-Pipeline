const NOW=Date.now(), H=3600000;

const demoCandidates=[

  {id:6,name:'Leon Bauer',    role:'Operations Manager',company:'Vitesse',   stage:'second',cvDate:'7 Apr 2026', interviewTs:NOW+72*H},

  {id:8,name:'Raj Patel',     role:'Head of Growth',    company:'Acme Corp', stage:'final', cvDate:'2 Apr 2026', interviewTs:NOW+52*H},

];


const avColors=[

  {bg:'#EDE9FE',fg:'#5B21B6'},{bg:'#FCE7F3',fg:'#9D174D'},

  {bg:'#DCFCE7',fg:'#166534'},{bg:'#FEF3C7',fg:'#92400E'},

  {bg:'#DBEAFE',fg:'#1E40AF'},{bg:'#FFE4E6',fg:'#9F1239'},

  {bg:'#F0FDF4',fg:'#14532D'},{bg:'#FFF7ED',fg:'#7C2D12'},

];


let candidates=[...demoCandidates];

const state={};

function initState(list){list.forEach((c,i)=>{if(!state[c.id])state[c.id]={prepped:false,confirmed:false,note:'',avColor:avColors[i%avColors.length]};});}

initState(candidates);


function ini(n){return n.split(' ').map(x=>x[0]).join('');}

function fmtTs(ts){

  const d=new Date(ts),hrs=(ts-NOW)/H;

  const t=d.toLocaleTimeString('en-GB',{hour:'2-digit',minute:'2-digit'});

  if(hrs<2)return 'Very soon';

  if(hrs<24)return t+' today';

  return d.toLocaleDateString('en-GB',{weekday:'short',day:'numeric',month:'short'})+' · '+t;

}

function u24(ts){return ts&&(ts-NOW)>0&&(ts-NOW)<24*H;}


function checkIcon(done){

  return done

    ?`<svg viewBox="0 0 10 10" fill="none" stroke="#166534" stroke-width="2.2"><polyline points="2,5 4,7 8,3"/></svg>`

    :`<svg viewBox="0 0 10 10" fill="none" stroke="#991B1B" stroke-width="1.8"><line x1="3" y1="3" x2="7" y2="7"/><line x1="7" y1="3" x2="3" y2="7"/></svg>`;

}


function calIcon(){

  return `<svg viewBox="0 0 11 11" fill="none" stroke="currentColor" stroke-width="1.4"><rect x="0.7" y="1.5" width="9.6" height="8.5" rx="1.5"/><line x1="3.5" y1="0.5" x2="3.5" y2="2.5"/><line x1="7.5" y1="0.5" x2="7.5" y2="2.5"/><line x1="0.7" y1="4.5" x2="10.3" y2="4.5"/></svg>`;

}


function renderStats(){

  const total=candidates.length;

  const prepped=candidates.filter(c=>state[c.id]&&state[c.id].prepped).length;

  const interviews=candidates.filter(c=>c.interviewTs).length;

  const confirmed=candidates.filter(c=>c.interviewTs&&state[c.id]&&state[c.id].confirmed).length;

  const urgent=candidates.filter(c=>c.interviewTs&&u24(c.interviewTs)&&state[c.id]&&!state[c.id].confirmed).length;

  document.getElementById('stats').innerHTML=`

    <div class="stat"><div class="stat-val">${total}</div><div class="stat-label">Total candidates</div></div>

    <div class="stat"><div class="stat-val">${prepped}/${total}</div><div class="stat-label">Prepped</div></div>

    <div class="stat"><div class="stat-val">${confirmed}/${interviews}</div><div class="stat-label">Interviews confirmed</div></div>

    <div class="stat urgent"><div class="stat-val">${urgent}</div><div class="stat-label">Under 24hrs unconfirmed</div></div>`;

}


function renderBanner(){

  const urgent=candidates.filter(c=>c.interviewTs&&u24(c.interviewTs)&&state[c.id]&&!state[c.id].confirmed);

  const b=document.getElementById('banner');

  if(urgent.length){

    b.style.display='flex';

    document.getElementById('banner-text').textContent=urgent.length===1

      ?urgent[0].name+' has an interview in under 24hrs and is not yet confirmed!'

      :urgent.length+' candidates have interviews under 24hrs — not yet confirmed!';

  } else b.style.display='none';

}


function render(){

  renderStats();

  renderBanner();

  const board=document.getElementById('board');

  board.innerHTML='';


  stages.forEach(stage=>{

    const sc=candidates.filter(c=>c.stage===stage.id);

    const col=document.createElement('div');

    col.className='col';

    col.innerHTML=`<div class="col-header">

      <div class="col-label-wrap">

        <div class="col-pip" style="background:${stage.color}"></div>

        <span class="col-label">${stage.label}</span>

      </div>

      <span class="col-count">${sc.length}</span>

    </div>`;


    sc.forEach(c=>{

      const s=state[c.id]||(state[c.id]={prepped:false,confirmed:false,note:'',avColor:avColors[0]});

      const hasInt=!!c.interviewTs;

      const isU24=u24(c.interviewTs);

      const stObj=stages.find(st=>st.id===c.stage);

      const av=s.avColor;


      const card=document.createElement('div');

      card.className='card';


      const accent=document.createElement('div');

      accent.className='card-accent';

      accent.style.background=stObj.color;

      card.appendChild(accent);


      const body=document.createElement('div');

      body.className='card-body';


      const avRow=document.createElement('div');

      avRow.className='av-row';

      avRow.innerHTML=`<div class="av" style="background:${av.bg};color:${av.fg}">${ini(c.name)}</div>

        <div style="min-width:0">

          <div class="cname">${c.name}</div>

          <div class="crole">${c.role} <em>at</em> <strong>${c.company}</strong></div>

        </div>`;

      body.appendChild(avRow);


      const chips=document.createElement('div');

      chips.className='chips';

      if(hasInt){

        chips.innerHTML=`<span class="chip chip-int">${calIcon()} ${fmtTs(c.interviewTs)}</span>`;

        if(isU24&&!s.confirmed) chips.innerHTML+=`<span class="chip chip-urgent">Under 24hrs</span>`;

      } else {

        chips.innerHTML=`<span class="chip chip-cv">CV sent ${c.cvDate}</span>`;

      }

      body.appendChild(chips);

      card.appendChild(body);


      const checks=document.createElement('div');

      checks.className='checks';


      const b1=document.createElement('div');

      b1.className='chk-block '+(s.prepped?'grn':'red');

      b1.innerHTML=`<div class="chk-icon ${s.prepped?'grn':'red'}">${checkIcon(s.prepped)}</div>

        <div class="chk-content">

          <div class="chk-title ${s.prepped?'grn':'red'}">${s.prepped?'Prep complete':'Prep needed'}</div>

          <div class="chk-desc${s.prepped?' thru':''}">Prepped & prep email sent</div>

        </div>`;

      const cb1=document.createElement('input');

      cb1.type='checkbox';cb1.className='chk-cb'+(s.prepped?'':' red');cb1.checked=s.prepped;

      cb1.addEventListener('change',()=>{state[c.id].prepped=cb1.checked;render();});

      b1.appendChild(cb1);

      checks.appendChild(b1);


      if(hasInt){

        const b2=document.createElement('div');

        b2.className='chk-block '+(s.confirmed?'grn':'red');

        b2.innerHTML=`<div class="chk-icon ${s.confirmed?'grn':'red'}">${checkIcon(s.confirmed)}</div>

          <div class="chk-content">

            <div class="chk-title ${s.confirmed?'grn':'red'}">${s.confirmed?'All confirmed':'Not confirmed'}</div>

            <div class="chk-desc${s.confirmed?' thru':''}">Confirmed, locked in & good luck sent</div>

        cb2.type='checkbox';cb2.className='chk-cb'+(s.confirmed?'':' red');cb2.checked=s.confirmed;

        b2.appendChild(cb2);

        if(!s.confirmed&&isU24){

      na.appendChild(nl);na.appendChild(pre);na.appendChild(ta);na.appendChild(sb);

      card.appendChild(na);

    });

}


function saveLoxo(){

  const key=document.getElementById('api-key-input').value.trim();

  const slug=document.getElementById('slug-input').value.trim();

  const domain=document.getElementById('domain-input').value.trim();

  if(!key||!slug||!domain){alert('Please fill in all three fields.');return;}

  localStorage.setItem('loxo_key',key);

  localStorage.setItem('loxo_slug',slug);

  localStorage.setItem('loxo_domain',domain);

  document.getElementById('modal').classList.remove('open');

  document.getElementById('api-badge').textContent='Loxo connected';
